pipeline {
    agent any

    stages {
        stage('Pull Latest Code') {
            steps {
                checkout scm
            }
        }

        stage('Run Unit Tests') {
            steps {
                bat 'pip install -r requirements.txt'
                bat 'pytest'
            }
        }

        stage('Build Docker Image') {
            steps {
                bat 'docker build -t sample-app .'
            }
        }

        stage('Push Docker Image') {
            steps {
                echo 'Docker Hub push will be configured later'
            }
        }
    }
}
